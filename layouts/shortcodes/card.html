{{- /* Link preview card: fetches the target URL at build time and extracts its OpenGraph metadata. */ -}} {{- $url := .Get 0 -}} {{- /* Defaults so the card degrades to a plain link if the fetch fails */ -}} {{- $title := $url -}} {{- $description := "" -}} {{- $image := "" -}} {{- $host := "" -}} {{- with urls.Parse $url -}}{{ $host = .Host }}{{- end -}} {{- $res := "" -}} {{- with try (resources.GetRemote $url) -}} {{- with .Err -}} {{- warnf "card: error fetching %q: %s" $url . -}} {{- else with .Value -}} {{- $res = . -}} {{- else -}} {{- warnf "card: no resource for %q" $url -}} {{- end -}} {{- end -}} {{- with $res -}} {{- $html := .Content -}} {{- with (findRESubmatch `]+property="og:title"[^>]+content="([^"]*)"` $html 1) -}} {{- with index . 0 -}}{{ $title = htmlUnescape (index . 1) }}{{- end -}} {{- end -}} {{- with (findRESubmatch `]+property="og:description"[^>]+content="([^"]*)"` $html 1) -}} {{- with index . 0 -}}{{ $description = htmlUnescape (index . 1) }}{{- end -}} {{- end -}} {{- with (findRESubmatch `]+property="og:image"[^>]+content="([^"]*)"` $html 1) -}} {{- with index . 0 -}}{{ $image = index . 1 }}{{- end -}} {{- end -}} {{- end -}}
